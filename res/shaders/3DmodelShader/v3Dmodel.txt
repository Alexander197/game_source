#version 460
layout(location = 0) in vec3 vertex_position;
layout(location = 1) in vec2 texture_coords;
layout(location = 2) in vec3 normals_coords;

out vec2 texCoords;

out vec3 Normal;
out vec3 FragPos;

uniform mat4 projectionMat;
uniform mat4 viewMat;
uniform mat4 modelMat;

uniform mat3 nModelMat;

void main() {
   texCoords = texture_coords;
   
   Normal = nModelMat * normals_coords;
   FragPos = vec3(modelMat * vec4(vertex_position, 1.0));
   
   gl_Position = projectionMat * viewMat * modelMat * vec4(vertex_position, 1.0);
}
